# FlowCollector 재생성 (eBPF privileged 활성화)
oc delete flowcollector cluster

cat << 'EOF' | oc apply -f -
apiVersion: flows.netobserv.io/v1beta2
kind: FlowCollector
metadata:
  name: cluster
spec:
  namespace: netobserv
  deploymentModel: Direct
  agent:
    type: eBPF
    ebpf:
      sampling: 50
      privileged: true
      interfaces: []
      excludeInterfaces:
      - lo
      resources:
        requests:
          memory: 50Mi
          cpu: 100m
        limits:
          memory: 800Mi
  processor:
    resources:
      requests:
        memory: 100Mi
        cpu: 100m
      limits:
        memory: 800Mi
  loki:
    enable: false
  consolePlugin:
    enable: false
EOF
